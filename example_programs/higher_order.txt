(* higher_order.txt *)
(* Demonstrates passing a function as an argument to another function. *)
(* expected out:
14
*)

(* combine accepts functions f and g as args *)
let combine = fun(f, g, x) ->
    f(g(x))
end;

let add_three = fun(x) ->
    x + 3
end;

let add_five = fun(x) -> 
    x + 5
end;

print(combine(add_three, add_five, 6));
